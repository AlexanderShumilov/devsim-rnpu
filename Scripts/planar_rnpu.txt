Set name="RNPU_500Ch_Ph1e15_SIMSGFITimplB_holeinjcontact_300K"

####### PROCESS SIMULATION ####### 
GO victoryprocess simflags="-P 4"

####### Parameters #######
set etchdepth = 0          # micron
set nativeoxide_thickness = 0.003 # micron
set metalwidth = 0.1# micron
set channellength = 0.5 # micron
set wf = 5.25 # 4.17 eV for electron injection or 5.25 eV for hole injection
set depth = 2       # micron
set temp = 300 # K

set width_inclsideox = $metalwidth+$nativeoxide_thickness
set effective_metalwidth = $metalwidth-$nativeoxide_thickness
set size_simwindow = $metalwidth+$channellength/2

# ---------- Mesh resolution (refined) ----------
set resManual = 0.002                 # 2 nm local refinement target
set meshx_metal = $metalwidth/20      # 5 nm
set meshx_channel = $channellength/100 # 10 nm
set meshz_etchdepth = 0.002           # 2 nm near surface
set meshz_depth = 0.02                # 20 nm in bulk

# ---------- Initialization ----------
INIT meshdepth=2 orientation=100 from=0.0 to=$size_simwindow depth=$depth gasheight=2.35 resolution=0.01,0.01 material=silicon dopants="phosphorus" dopingvalues="1e15"

# ---------- Z-direction mesh (vertical) ----------
# Fine mesh near surface and implant region
set z_surface_fine = 0.0
set z_channel_top  = 0.05 * $depth        ;# top ~5% of silicon depth
set z_boron_end    = 0.2 * $depth         ;# where boron tail ends
set z_bulk         = $depth

set dz_surface     = 0.002                ;# 2 nm fine at surface
set dz_channel     = 0.005               ;# 5 nm in channel region
set dz_boron       = 0.02                  ;# 20 nm in boron tail
set dz_bulk        = 0.1                  ;# coarser in bulk

LINE Z LOCATION=$z_surface_fine SPACING=$dz_surface
LINE Z LOCATION=$z_channel_top  SPACING=$dz_channel
LINE Z LOCATION=$z_boron_end    SPACING=$dz_boron
LINE Z LOCATION=$z_bulk         SPACING=$dz_bulk

# ---------- X-direction mesh (lateral) ----------
# Fine mesh under metal and channel, coarse far away
set x_left_fine    = 0.0
set x_metal_right  = $metalwidth
set x_channel_end  = $metalwidth + $channellength
set x_far_right    = $size_simwindow

set dx_left_fine   = $metalwidth/20       ;# fine under metal
set dx_channel     = $channellength/50    ;# fine in channel
set dx_far         = $channellength/10    ;# coarser far from active area

LINE X LOCATION=$x_left_fine    SPACING=$dx_left_fine
LINE X LOCATION=$x_metal_right  SPACING=$dx_channel
LINE X LOCATION=$x_channel_end  SPACING=$dx_far
LINE X LOCATION=$x_far_right    SPACING=$dx_far


# ---------- Refinement boxes ----------
REFINEGEOMESH BOX BOX.FROM="0.0,-0.05" BOX.TO="$effective_metalwidth,0.05" APPLY_REFINEMENT LOCK RESOLUTION=0.002
REFINEGEOMESH BOX BOX.FROM="$metalwidth-0.01,-0.05" BOX.TO="$size_simwindow,0.05" APPLY_REFINEMENT LOCK RESOLUTION=0.002
REFINEGEOMESH BOX BOX.FROM="$metalwidth,$etchdepth-0.05" BOX.TO="$width_inclsideox,0.05" APPLY_REFINEMENT LOCK RESOLUTION=0.002
REFINEGEOMESH BOX BOX.FROM="0.0,-$depth" BOX.TO="$size_simwindow,-0.2" APPLY_REFINEMENT RESOLUTION=0.02

# ---------- Implantation ----------
# localized boron/arsenic implant using fitted Gaussian profile (select between boron / arsenic)
# DOSE in cm^-2, PEAK and SIGMA in um
MASK BETWEEN.X="$metalwidth, $size_simwindow"
DOPING SILICON BORON DOSE=6e13 PEAK=2.927699e-03 GAUSS.SIGMA=2.766469e-02 LATERAL.SIGMA=$metalwidth/6 # BORON
#DOPING SILICON ARSENIC DOSE=6.5e13 PEAK=1.317234e-03 GAUSS.SIGMA=2.7181546e-02 LATERAL.SIGMA=$metalwidth/6 # ARSENIC
STRIP

# ---------- Metal stack ----------
DEPOSIT left.to=$metalwidth material=titanium thick=0.001 directional
DEPOSIT left.to=$metalwidth material=palladium thick=0.025 directional

# ---------- Native oxide ----------
DEPOSIT BETWEEN.X=$metalwidth, $size_simwindow material=oxide thick=$nativeoxide_thickness min

SAVE name=$'name'

####### STRUCTURE REMESHING #######
GO victorymesh simflags="-P 4"
LOAD in="$'name'"
MIRROR axes="+1x"
REMESH conformal 
SAVE out="$'name'"

####### DEVICE SIMULATION #######
GO victoryd simflags="-P 2"

MESH inf=$'name'.str no.contact width=1 # 1 um is default size of device; simply scales current

set size_real_simwindow = 2*$size_simwindow
ELECTRODE coord.region="0,-0.01" name=source
ELECTRODE coord.region="$size_real_simwindow,-0.01" name=drain
# note substrate is floating!! Can also play with this as often tied to supply voltage / ground

CONTACT NAME="source" workfunction=$wf # alternatively neutral contact
CONTACT NAME="drain" workfunction=$wf

MODELS KLASSEN KLASRH CVT temp=$temp
# MOB.INCOMPL
MATERIAL MATERIAL=silicon NC300=2.752e19 NV300=1.021e19 MUN=1000 MUP=500 EG300=1.08 TAUN0=1.0e-7 TAUP0=1.0e-7
# note these are the default silicon material parameters

#MOBILITY CVT.N ^CVT.SR.N ^CVT.SP.N CVT.P ^CVT.SR.P ^CVT.SP.P

SOLVE init
LOG outfile=$'name'_1.log
OUTPUT BAND.PARAM CHARGE TRAPS E.MOBILITY H.MOBILITY QFN QFP

# Can freely change the measurement conditions
#SOLVE name=drain vdrain=-10 vstep=0.1 vfinal=-1
#SAVE outfile=$'name'_neg1V.str
#SOLVE name=drain vdrain=-1 vstep=0.1 vfinal=0
#SAVE outfile=$'name'_0V.str
#SOLVE name=drain vdrain=0 vstep=0.001 vfinal=0.005
#SAVE outfile=$'name'_10mV.str
SOLVE name=drain vdrain=-0.001 vstep=0.001 vfinal=0
SAVE outfile=$'name'_0mV.str
SOLVE name=drain vdrain=0 vstep=0.001 vfinal=0.01
SAVE outfile=$'name'_10mV.str
SOLVE name=drain vdrain=0.011 vstep=0.001 vfinal=0.1
SAVE outfile=$'name'_100mV.str
SOLVE name=drain vdrain=0.101 vstep=0.001 vfinal=1
SAVE outfile=$'name'_1V.str
SOLVE name=drain vdrain=1.01 vstep=0.01 vfinal=10
SAVE outfile=$'name'_10V.str
#SOLVE name=drain vdrain=10.1 vstep=0.1 vfinal=100
#SAVE outfile=$'name’_100V.str
LOG off
SAVE outfile=$'name'_1.str

####### POST-PROCESSING #######
GO victoryextract
LOAD in=$'name'_1.log
CURVE name="IVlogscale" x=LOG10(Drain Int. Voltage) y=LOG10(Drain Current)
CURVE name="slopeplot" expression=DERIVATIVE(curve.IVlogscale)
EXTRACT.CSV csvfile="$'name'_slopeplot"

victoryvisual
